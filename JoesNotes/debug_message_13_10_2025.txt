using Microsoft.Extensions.FileProviders;
using System.IO;

var builder = WebApplication.CreateBuilder(args);

{
    options.ListenLocalhost(5000);
});

var currentDir = Directory.GetCurrentDirectory();
var projectRoot = Path.GetFullPath(Path.Combine(currentDir, "..", "..", "..")); 

var dashboardPath = Path.Combine(projectRoot, "Dashboard");
var resourcesPath = Path.Combine(projectRoot, "Resources");
var killScriptPath = Path.Combine(projectRoot, "kill_protocol.sh");

app.UseStaticFiles(new StaticFileOptions
{
    FileProvider = new PhysicalFileProvider("/home/vboxuser/Desktop/MyKY-main/Dashboard"),
    RequestPath = "/dashboard"
});

app.UseStaticFiles(new StaticFileOptions
{
    FileProvider = new PhysicalFileProvider("/home/vboxuser/Desktop/MyKY-main/Resources"),
    RequestPath = "/resources"
});

app.MapGet("/", () => Results.Redirect("/dashboard/dashboard.html"));
app.MapGet("/dashboard", () => Results.Redirect("/dashboard/dashboard.html"));

app.MapPost("/kill-app", () =>
{
    try
    {
        var process = new System.Diagnostics.Process
        {
            StartInfo = new System.Diagnostics.ProcessStartInfo
            {
                FileName = "bash",
                Arguments = killScriptPath,
                UseShellExecute = false,
                CreateNoWindow = true
            }
        };
        process.Start();
        process.WaitForExit();

        return Results.Ok("Kill signal sent successfully");
    }
    catch (System.Exception ex)
    {
        return Results.Problem($"Failed to execute kill script: {ex.Message}");
    }
});

app.Run();
